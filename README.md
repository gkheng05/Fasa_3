```
    $hostname = "127.0.0.1"; //default if using xampp
    $username = "user"; //mysql username
    $password = "pass"; //mysql password
    $dbName = "db"; //mysql database name
    $dbManager = new dbManager($hostname, $username, $password, $dbName);
```
